Comparação de maior menor ou igual


            let one = 1
            let two = 2

            // > Maior que
            console.log(one > two) false

            // >= Maior ou igual a 
            console.log(one >= 1) true
            console.log(two >= 1) false

            // < Menor que
            console.log(one < two) true

            // <= Menor ou igual a 
            console.log(one <= two) true
